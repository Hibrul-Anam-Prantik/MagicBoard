import java.io.*;
import java.util.*;

public class Task08 {

    static int postIndex;
    static int[] inOrder;
    static int[] postOrder;

    public static class Node {
        int val;
        Node left;
        Node right;

        Node(int v) {
            val = v;
        }
    }

    public static Node buildPreBT(int start, int end, HashMap<Integer, Integer> inMap) {
        if (start > end) return null;
        int rootval = postOrder[postIndex--];
        Node root = new Node(rootval);
        int index = inMap.get(rootval);

        root.right = buildPreBT(index + 1, end, inMap);
        root.left = buildPreBT(start, index - 1, inMap);

        return root;
    }

    public static void preOrderTraversal(Node root) {

        if (root == null) return;
        System.out.print(root.val + " ");
        preOrderTraversal(root.left);
        preOrderTraversal(root.right);


    }

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int N = Integer.parseInt(br.readLine().trim());
        inOrder = new int[N];
        postOrder = new int[N];
        postIndex = N - 1;

        String[] inBT = br.readLine().trim().split(" ");
        for (int i = 0; i < N; i++) {
            inOrder[i] = Integer.parseInt(inBT[i]);
        }
        String[] postBT = br.readLine().trim().split(" ");
        for (int i = 0; i < N; i++) {
            postOrder[i] = Integer.parseInt(postBT[i]);
        }

        HashMap<Integer, Integer> inMap = new HashMap<>();
        for (int i = 0; i < N; i++) {
            inMap.put(inOrder[i], i);
        }

        Node root = buildPreBT(0, N - 1, inMap);

        preOrderTraversal(root);
        System.out.println();

    }


}
