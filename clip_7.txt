import java.io.*;
import java.util.*;
 
public class Main {
 
    static int preIndexTracker;
    static int[] inorderSequence;
    static int[] preorderSequence;
 
     public static class TreeNode {
        int value;
        TreeNode leftNode;
        TreeNode rightNode;
 
        TreeNode(int v){
            value = v;
        }
    }
    
    public static TreeNode constructTree(int start, int end, HashMap<Integer, Integer> inorderMap) {
        if(start > end) return null;
        
        int rootValue = preorderSequence[preIndexTracker++];
        TreeNode root = new TreeNode(rootValue);
        int rootIndexInorder = inorderMap.get(rootValue);
        
        root.leftNode = constructTree(start, rootIndexInorder - 1, inorderMap);
        root.rightNode = constructTree(rootIndexInorder + 1, end, inorderMap);
 
        return root;
    }
 
    public static void printPostorder(TreeNode currentRoot) {
      
        if(currentRoot == null) return;
        
        printPostorder(currentRoot.leftNode);
        printPostorder(currentRoot.rightNode);
        System.out.print(currentRoot.value + " ");
    }
 
    public static void main(String[] args) throws IOException {
        
        BufferedReader inputReader = new BufferedReader(new InputStreamReader(System.in));
        
        int treeSize = Integer.parseInt(inputReader.readLine().trim());
        
        inorderSequence = new int[treeSize];
        preorderSequence = new int[treeSize];
 
        String[] inorderTokens = inputReader.readLine().trim().split(" ");
        for(int i = 0; i < treeSize; i++) {
            inorderSequence[i] = Integer.parseInt(inorderTokens[i]);
        }
        
        String[] preorderTokens = inputReader.readLine().trim().split(" ");
        for(int i = 0; i < treeSize; i++) {
            preorderSequence[i] = Integer.parseInt(preorderTokens[i]);
        }
        
        HashMap<Integer, Integer> inorderPositionMap = new HashMap<>();
        for(int i = 0; i < treeSize; i++) {
            inorderPositionMap.put(inorderSequence[i], i);
        }
 
        preIndexTracker = 0;
        
        TreeNode treeRoot = constructTree(0, treeSize - 1, inorderPositionMap);
 
        printPostorder(treeRoot);
        System.out.println();
    }
}
